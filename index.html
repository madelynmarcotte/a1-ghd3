// script.js // Select the body element and append an SVG container const svg = d3.select('body').append('svg') .attr('width', 200) // Set the width of the SVG .attr('height', 100); // Set the height of the SVG // Append a rectangle to the SVG svg.append('rect') .attr('width', 150) // Set the width of the rectangle .attr('height', 50) // Set the height of the rectangle .attr('fill', 'green'); // Set the fill color to green
